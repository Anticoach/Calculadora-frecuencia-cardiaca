            <input type="number" step="0.01" id="intensity" required>

        </div>

        <button onclick="calculateHeartRate()">Calcular</button>

        <h3 id="result"></h3>

    </div>

    <script>

        function calculateHeartRate() {

            const age = document.getElementById('age').value;

            const rhr = document.getElementById('rhr').value;

            const intensity = document.getElementById('intensity').value;

            const maxHR = 220 - age;

            const karvonenHR = parseInt(rhr) + ((maxHR - rhr) * intensity);

            document.getElementById('result').innerText = `Fórmula Clásica: ${maxHR}, Fórmula Karvonen: ${karvonenHR.toFixed(2)}`;

        }

    </script>
